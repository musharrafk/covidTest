$(document).ready(function(){

	$('#mrfcandidateform1').submit(function(evt) {
             evt.preventDefault();
              $("#addCandMrf1").prop('disabled', true);
			 
            var Id = $('#id').val();
			var formData = new FormData(this);
			$("#page_loader1").show();
					//var formdata = $("#mrfcandidateform").serialize();
						$.ajax({
							type: 'POST',
							url: '<?php echo site_url();?>helpdesk/insert_reply_ticket',
							data: formData,
							cache:false,
                            contentType: false,
                            processData: false,
							success:function (data){
								if(data==1)
								{
									if($("#id").val()>0)
									{
										jQuery.noticeAdd({text:"added successfully"});
										$("#list").trigger("reloadGrid");
										editDialog.dialog("close");
									}
									else
									{
										jQuery.noticeAdd({text:"added successfully"});
										$("#list").trigger("reloadGrid");
										editDialog.dialog("close");                    
                   // $('#counterCandidate'+mrfId).html(parseInt($('#counterCandidate'+mrfId).html()) + 1);
									}
								}
								else
								{
									if($("#id").val()>0)
									{
										jQuery.noticeAdd({text:"Update Failed"});
									}
									else
									{
										jQuery.noticeAdd({text:"Added Failed"});
									}
								}
								$("#page_loader1").hide();
							  $("#addCandMrf1").prop('disabled', false);
							}
						});
			
	});
	
	//Call city With Ajax
    $("#currentState").change(function(){
        var selectedState = $("#currentState").val();		
	  if(selectedState!=''){		
		 $.post("<?php echo site_url("mrf/ajaxCity")?>",{sid:selectedState},

				//alert(data);

					$("#currentLocation").html(data);
					
					}
				});
			
		} else {
			$("#currentLocation").html('<option>Select Job Location</option>');
		 }
       });
	});
